### 문제5
# 회귀모형의 성능을 평가할 수 있는 지표: RMSE(평균 제곱근 오차(Root Mean Square Error; RMSE)는 추정 값 또는 모델이 예측한 값과 실제 환경에서 관찰되는 값의 차이를 다룰 때 흔히 사용하는 측도입니다.)

# train RMSE
salary_pred <- predict(Fitmodel, newdata = train_data)
salary_true = train_data$salary
RMSE(salary_pred, salary_true)
## [1] 21358.84
# test RMSE
salary_pred <- predict(Fitmodel, newdata = test_data)
salary_true = test_data$salary
RMSE(salary_pred, salary_true)
## [1] 24785.88
